示例


为了帮助您理解如何通过HTTPS操作Web API，请查看这些示例。在这些示例中实现以下操作方案：


- 创建MT5Request对象，并在其中实现Keep-Alive连接机制。

- 调用用于在服务器上进行验证的Auth命令。

- 然后实现GET/POST请求，将命令发送给服务器。

- 这些示例包含用于连接到平台的SSL证书的验证。当使用不受信任的证书的证书时，激活授权错误。您可以通过注释相应的代码字符串来禁用验证(参见注释)。


JavaScript


工作示例，安装[Node JS](https://nodejs.org/)。


```javascript
const https = require("https");

const crypto = require('crypto');

const buffer = require('buffer');



function MT5Request(server, port) {

 this.server = server;

 this.port = port;

 this.https = new https.Agent();

 this.https.maxSockets = 1; // 仅使用一个连接

}



MT5Request.prototype.Get = function (path, callback) {

 var options = {

 hostname:this.server,

 port:this.port,

 path: path,

 agent:this.https,

 headers: { "Connection":"keep-alive" },

 //rejectUnauthorized: false, //如果您使用自签名证书，请取消注释此行

 };

 var req = https.get(options, function (res) {

 res.setEncoding('utf8');

 var body = "";

 res.on('data', function (chunk) {

 body += chunk;

 });

 res.on('end', function () {

 callback(null, res, body);

 });

 });

 req.on('error', function (e) {

 console.log(e);

 return callback(e);

 });

};



MT5Request.prototype.Post = function (path, body, callback) {

 var options = {

 hostname:this.server,

 port:this.port,

 path: path,

 agent:this.https,

 method:"POST",

 headers: {

 "Connection":"keep-alive",

 "Content-Type":"application/x-www-form-urlencoded",

 "Content-Length": Buffer.byteLength(body),

 },

 rejectUnauthorized: false,// 如果使用自签名证书，请注释这一行。

 };

 var req = https.request(options, function (res) {

 res.setEncoding('utf8');

 var body = "";

 res.on('data', function (chunk) {

 body += chunk;

 });

 res.on('end', function () {

 callback(null, res, body);

 });

 });

 req.on('error', function (e) {

 console.log(e);

 return callback(e);

 });

 req.write(body);

 req.end();

};





MT5Request.prototype.ParseBodyJSON = function (error, res, body, callback) {

 if (error) {

 callback && callback(error);

 return (null);

 }

 if (res.statusCode != 200) {

 callback && callback(res.statusCode);

 return (null);

 }

 var answer = null;

 try {

 answer = JSON.parse(body);

 }

 catch {

 console.log("Parse JSON error");

 }

 if (!answer) {

 callback && callback("invalid body answer");

 return (null);

 }

 var retcode = parseInt(answer.retcode);

 if (retcode != 0) {

 callback && callback(answer.retcode);

 return (null);

 }

 return (answer);

}



MT5Request.prototype.ProcessAuth = function (answer, password) {

 //---

 var pass_md5 = crypto.createHash('md5');

 var buf = buffer.transcode(Buffer.from(password, 'utf8'), 'utf8', 'utf16le');

 pass_md5.update(buf, 'binary');

 var pass_md5_digest = pass_md5.digest('binary');

 //---

 var md5 = crypto.createHash('md5');

 md5.update(pass_md5_digest, 'binary');

 md5.update('WebAPI', 'ascii');

 var md5_digest = md5.digest('binary');

 //---

 var answer_md5 = crypto.createHash('md5');

 answer_md5.update(md5_digest, 'binary');

 var buf = Buffer.from(answer.srv_rand, 'hex');

 answer_md5.update(buf, 'binary');

 //---

 return (answer_md5.digest('hex'));

}



MT5Request.prototype.ProcessAuthFinal = function (answer, password, cli_random) {

 //---

 var pass_md5 = crypto.createHash('md5');

 var buf = buffer.transcode(Buffer.from(password, 'utf8'), 'utf8', 'utf16le');

 pass_md5.update(buf, 'binary');

 var pass_md5_digest = pass_md5.digest('binary');

 //---

 var md5 = crypto.createHash('md5');

 md5.update(pass_md5_digest, 'binary');

 md5.update('WebAPI', 'ascii');

 var md5_digest = md5.digest('binary');

 //---

 var answer_md5 = crypto.createHash('md5');

 answer_md5.update(md5_digest, 'binary');

 answer_md5.update(cli_random, 'binary');

 return (answer.cli_rand_answer == answer_md5.digest('hex'));

}



MT5Request.prototype.Auth = function (login, password, build, agent, callback) {

 if (!login || !password || !build || !agent)

 return;

 var self = this;

 self.Get("/api/auth/start?version=" + build + "&agent=" + agent + "&login=" + login + "&type=manager", function (error, res, body) {

 var answer = self.ParseBodyJSON(error, res, body, callback);

 if (answer) {

 var srv_rand_answer = self.ProcessAuth(answer, password);

 var cli_random_buf = crypto.randomBytes(16);

 cli_random_buf_hex = cli_random_buf.toString('hex');

 self.Get("/api/auth/answer?srv_rand_answer=" + srv_rand_answer + "&cli_rand=" + cli_random_buf_hex, function (error, res, body) {

 var answer = self.ParseBodyJSON(error, res, body, callback);

 if (answer) {

 if (self.ProcessAuthFinal(answer, password, cli_random_buf))

 callback && callback(null);

 else

 callback && callback("invalid final auth answer");

 }

 });

 }

 });

 return (true);

};



// 使用示例

var req = new MT5Request("my.broker.com", 443);

// 使用“验证”命令在服务器上进行身份验证

req.Auth(1000,"Password", 1985,"WebManager", function (error) {

 if (error) {

 console.log(error);

 return;

 }

 // 让我们使用symbol_get命令请求名为TEST的交易品种

 req.Get("/api/symbol/get?symbol=TEST", function (error, res, body) {

 if (error) {

 console.log(error);

 return;

 }

 var answer = req.ParseBodyJSON(error, res, body,null);

 var symbol = answer.answer;

 console.log(symbol);

 // 更改描述

 symbol.Description = 'My Description';

 // 使用symbol_add命令将更改发送至服务器

 req.Post('/api/symbol/add', JSON.stringify(symbol), function (error, res, body) {

 if (error) {

 console.log(error);

 return;

 }

 var answer = req.ParseBodyJSON(error, res, body,null);

 var symbol = answer.answer;

 console.log(symbol);

 });

 });

});
```

命令格式


Web API命令使用标准的POST和GET请求进行传递：


- 命令名及其参数完全符合Web API协议。

- 命令参数在?之后并且用&分隔。

- 其他请求体以JSON格式传递。

- 与访问服务器公共点连接的域指定为主机。

- 请求中的内容类型：application/x-www-form-urlencoded。


该命令可以用示意图表示如下：


```
GET /command?param1=value1&param2=value2&...

Host: {domain bound to public address}

Accept: */*

User-Agent:MetaTrader 5 Web API/5.2005 (Windows NT 6.2; x64)

Connection: keep-alive

内容-类型：application/x-www-form-urlencoded

{

...

}



POST /command?param1=value1&param2=value2&...

Host: {domain bound to public address}

Accept: */*

User-Agent:MetaTrader 5 Web API/5.2005 (Windows NT 6.2; x64)

Connection: keep-alive

内容-类型：application/x-www-form-urlencoded

{

...

}
```


在请求中指定以下数据：


- Host — 连接访问服务器公共点的域。

- command — 命令名称

- param1, value1, ... — 如果用于此命令的附加参数。

- { ... } — 如果为指定命令提供了附加请求，则其正文将是 JSON 格式的文本。例如，当使用/api/user/add命令时，创建的账户参数以附加主体传递。
一些参数区分大小写，例如，交易账户的 LeadSource和LeadCampaign 字段。因此，JSON正文中的字段名称必须始终按照文档中指定的大小写传递。例如，条目{ ..."LEADCAMPAIGN":"website_ad" ... }不正确。正确的规范是 { ..."LeadCampaing":"website_ad" ... }。


来自服务器的响应以JSON格式传递。响应代码通常传递为第一个参数，之后跟随附加信息。


> { "retcode" : "retcode", "answer" : { "param1" : "value1", "param2" : "value2", ... } }


例如，响应[/api/user/get](webapi_user_get.htm)命令时，您将收到关于所请求用户的响应代码和数据。


> 命令、参数和服务器响应都以Unicode (UTF-8)格式传递。 在身份验证期间和之后，所有命令都应该在一个时段中传递。

主接口(Rest API)


MetaTrader 5 Web API执行经理连接至交易服务器的以下命令：


- 配置数据库

- 交易

- 用户

- 客户

- 邮件

- 新闻

- 价格

- 每日报告

- 设置文件

- 订阅


> 使用经理命令的信息访问由针对用于连接到服务器的经理账户设置的访问权限来完整定义。


使用经理接口的主要方法是通过GET和POST请求(Rest API)发送命令。然而，MetaTrader 5 SDK程序包还包括此协议的 [PHP](php.htm) 和 [.NET](net.htm) 实施示例。

保持连接


访问服务器作为一种‘stateful’服务运行：您应首先进行[验证](webapi_rest_authentication.htm)，然后保持连接直至您完成操作（持续连接）。如果连接丢失，则需要重新验证。


在传递的所有命令中，连接类型应该显示为"keep-alive"：


```
GET /command?param1=value1&param2=value2&...

Host: {domain bound to public address}

Accept: */*

User-Agent:MetaTrader 5 Web API/5.2005 (Windows NT 6.2; x64)

Connection: keep-alive

内容- 类型：application/x-www-form-urlencoded

{

...

}
```


如果180秒内未接收到来自客户的数据包，服务器会断开连接。由此，将不可能进一步执行命令，直到您完成 [身份验证](webapi_rest_authentication.htm)过程。


为了维持连接至服务器，Web客户必须将空数据包（称为“ping”）发送到服务器。发送ping的最佳间隔时间是 20 秒。您不应太频繁地发送ping。


在Rest API中，您可以将 [/api/test/access](webapi_pings.htm#test_access) 查询用于这些目标。


提供了用于保持PHP和.NET implementations连接的特殊方法：


- MTWebAPI::Ping

- MTWebAPI.Ping


当使用文本API时，应该发送空数据包（数据包正文为空以及数据包接续标识为零）。数据包号码应在为客户命令 (0-3FFF) 指定的范围中。


客户空数据包的示例：


> 000000010 000000640


MetaTrader 5访问服务器也每20秒向客户自动发送空数据包。该等命令的号码在范围4000-7FFF中。


服务器空数据包的示例：


> 000042680 000062170


> 请注意，空命令的发送从服务器发送，不取决于对Web客户命令响应的发送。在一个混合流中接收这些数据包。


检查连接状态


要检查与平台访问服务器的Web客户连接，以及验证对已发送请求的响应，请发送/test_access命令。


Rest API


请求格式


> GET /test_access POST /test_access


响应格式


> { "retcode" : "xxxx yyyy" }


Raw API


请求格式


> TEST_ACCESS\r\n


响应格式


> TEST_ACCESS|RETCODE=xxxx yyyy|\r\n


响应参数


- retcode — 如果成功，则返回响应代码 0。否则，将返回错误代码。


退出操作


要终止Web客户和交易平台之间的连接，请发送/quit命令。


Rest API


请求格式


> GET /quit POST /quit


响应格式


> { "retcode" : "xxxx yyyy" }


Raw API


请求格式


> QUIT\r\n


响应格式


> QUIT|RETCODE=xxxx yyyy|\r\n

通过HTTPS设置操作


与交易平台的所有连接都可通过访问服务器建立。在访问服务器（HTTP未连接）上实现对HTTPS连接的支持。因此，它们可以充当web服务器。


若要启用通过HTTPS连接到访问服务器，那么至少将其一个[公共地址](https://support.metaquotes.net/zh/docs/mt5/platform/administration/admin_network/network_add_edit#public)与某个域相关联，即，将实现POST和GET请求的主机。为此，必须在DNS服务器上指定对应的条目。例如，如果您想通过主机 abc.broker.com实现操作，则应将以下条目添加到DNS："XXX.XXX.XXX.XXX abc.broker.com"，这里XXX.XXX.XXX.XXX就是访问服务器的公共地址。


因为只支持受保护的TLS连接(HTTPS)，因此必须在访问服务器上为对应的域安装SSL证书。使用MetaTrader 5 Administrator中的“集成\网络服务\SSL证书”部分来上传和管理证书：


这些设置应用于所有访问服务器。证书列表中的更改可立即应用，因此您无需重新启动服务器。


可以在一台服务器上支持来自多个域的连接（[SNI模式](https://ru.wikipedia.org/wiki/Server_Name_Indication)）。在这种情况下，必须为每个域安装对应的SSL证书。当连接时，客户端指定要连接的地址。访问服务器检查所有已安装的证书，并自动向客户端提供必要的证书：


- 首先，服务器试图为确切匹配的域找到证书。

- 如果没有，则会尝试使用掩码"*"找出最适合的证书。

- 如果没有找到证书，服务器将返回列表中的第一个证书。


> 当在Web API客户端使用自签名证书时，请将它们添加到受信任证书列表或者允许使用不受信任的证书。

协议扩展


MetaTrader 5 API 协议可扩展。除了标准的[经理端连接命令](webapi_main.htm)以外，Web 客户可以向服务器发送任何其他文本命令。然而，必须在服务器端上执行该等命令的处理。


可以使用通过 MetaTrader 5 Server API 创建的服务器插件处理自定义 Web API 命令。针对该目的，Server API 提供了特殊挂钩 [IMTCustomSink::HookWebAPICommand](imtcustomsink_hookwebapicommand.htm)。


考虑发送自定义命令的示例：


> POST /echo { “text”: “Request/Reply request”, “seq”: 1 }


该命令可以在插件端处理如下：


```
MTAPISTR cmd={},request{};

MTAPIRES retcode;

//--- 检查这是否是我们的命令

if(CMTStr::CompareNoCase(command,L"ECHO")!=0)

 return(MT_RET_OK_NONE);

//--- 检查内容

if(indata->WebReadCommand(cmd)!=MT_RET_OK)

 {

 //--- 错误

 return(MT_RET_ERR_DATA);

 }

//--- 阅读请求

retcode=indata->ReadStr(request);

if(retcode!=MT_RET_OK)

 {

 //--- 错误

 return(MT_RET_ERR_DATA);

 }

//--- 通过自定义参数处理自定义命令

if(CMTStr::Compare(request,L"{“text”: “Request/Reply request”, “seq”: 1 }")!=0)

 {

 //--- 错误

 return(MT_RET_ERR_DATA);

 }

//--- 发送响应

retcode=outdata->AddStr(L"{“text”: “Request/Reply reply”, “seq”: 1}");

if(retcode!=MT_RET_OK)

 {

 //--- 错误

 return(retcode);

 }

//--- 成功

return(MT_RET_OK);
```


该命令将返回以下命令，作为对该命令的响应：


> { "retcode" : "0 Done", "answer" : {“text”: “Request/Reply reply”, “seq”: 1} }


有关自定义命令处理的更多详情可在[IMTCustomSink::HookWebAPICommand](imtcustomsink_hookwebapicommand.htm)部分获得。


命令的保留名称


交易平台有一些保留使用的特殊名称。这些名称不能用于自定义命令：


- AUTH_*

- USER_*

- TRADE_*

- TIME_*

- GROUP_*

- ORDER_*

- POSITION_*

- DEAL_*

- HISTORY_*

- TICK_*

- MAIL_*

- NEWS_*

- COMMON_*

- QUIT*

- TEST_TRADE

- TEST_ACCESS


> 扩展协议时，建议您使用不会与在 MetaTrader 5 服务器端或者任何其他第三方插件上执行的命令交叉的命令名称。例如，您可以在命令名称中使用公司名称: MYCOMPANY_MYCOMMAND。

| | Web API / 身份验证 | |
| --- | --- | --- |






身份验证


要向服务器发送命令，客户应在那里通过身份验证。这两个命令用于：


- /auth_start — 身份验证开始命令。

- /auth_answer — 身份验证请求响应命令。


身份验证分为几个步骤，借此服务器和客户通过不同参数发送这些命令。


- 开始身份验证的请求

- 开始在访问服务器上进行身份验证

- 响应服务器

- 身份验证结束


JavaScript和PHP的身份验证示例提供在[示例](webapi_examples.htm)部分。


开始身份验证的请求


客户应该发送请求以使用/auth_start命令开始验证，例如：


> GET https://server.address:443/auth_start?version=484&agent=test&login=14&type=manager


在命令中传递以下用"&"字符隔开的参数：


- version — 网页客户端的版本。使用这个字段来管理您的应用程序版本。当网页客户端连接到平台时，指定的版本将被记录到访问服务器日志。例如，"1000"：登录名(Manager WebAPI build 3470)。为了方便起见，您可以在这里指定当前的平台版本。如果您不需要版本管理，可以输入任何不小于484的数字。

- agent — Web代理的任意名字。名字不应为空。名称不应为空且不应包含空格。

- login — 应该根据该账户的登录名执行验证。

- type — 账户类型。目前只支持一个值，就是'manager'。这意味着使用经理账户连接。使用的账户完全决定连接权限，包括允许连接的IP地址列表。


> 验证开始请求(/api/auth/start)和向服务器发送响应(/api/auth/answer)之间的时间不应超过10秒。否则，身份验证将失败。


开始在访问服务器上进行身份验证


响应身份验证开始命令时，访问服务器发送包含响应代码和随机序列的响应。


> { "retcode" : "0 Done", "version_access" : "1290", "srv_rand" : "d4e005317e38bc0c349a51a0d73a07eb" }


在命令中传递以下用"|"字符隔开的参数：


- retcode — 响应代码及其字符串描述。0 意味着成功完成。

- version_access — 用于连接至交易服务器的访问服务器的版本；

- srv_rand — 由服务器生成的Hex格式的随机16 字节序列。发送响应时该序列应该应用密码。


响应服务器


接收随机序列之后，客户应该向其应用密码（密码散列）并使用/auth_answer命令将其发送到服务器：


> GET https://server.address:443/auth_answer?srv_rand_answer=8b67609a264764a528ec8c709ab71df2&cli_rand=34b6bccab8bfebdf462d1312043df1cc


在命令中传递以下用"&"字符隔开的参数：


- srv_rand_answer — MD5随机序列散列和密码散列。

- cli_rand — 您应该自己生成一个用于服务器身份验证的随机序列。序列应该是16 字节大小的Hex格式。


密码散列


密码散列计算方式如下：


> Password hash=MD5(MD5('Password')+'WebAPI')


WebAPI 的六个字符被添加到 MD5 密码散列中，并再次从结果获得 MD5 散列。然后，发送到服务器时，获得的散列再次结合服务器的随机顺序。散列取自结果值，然后其被发送到服务器。这样可确保授权的最大安全性。


构成srv_rand_answer的示例：


```
password hash=MD5(MD5('Password1')+'WebAPI')=904ba8ecb16273d2f0ae9c3b8a023752

srv_rand=73007dc7184747ce0f7c98516ef1c851

//--- 计算公式：srv_rand_answer=MD5(password hash+srv_rand)

srv_rand_answer=MD5(904ba8ecb16273d2f0ae9c3b8a023752+73007dc7184747ce0f7c98516ef1c851)=77fe51827f7fa69dd80fbec9aa33f1bb
```


如何构成srv_rand_answer的逐步讲解的示例：


让我们来分析Web password=Password1和srv_rand="73007dc7184747ce0f7c98516ef1c851"。


1.我们用 Unicode (UTF-16-LE)计算密码的MD5。在这种情况下，unicode字符串Password1：


> MD5('Password1')


结果作为字节：


> [ 236,46,146,255,112,191,70,145,174,12,104,224,47,220,108,34 ]


2.将单字节WebAPI字符串作为6个字节(UTF-8)添加到字节结果：


> MD5('Password1')+'WebAPI'


'WebAPI’ 被表现为 = [ 87,101,98,65,80,73 ]。结果作为字节：


> [ 236,46,146,255,112,191,70,145,174,12,104,224,47,220,108,34,87,101,98,65,80,73 ]


3. 获得密码的最终散列作为来自步骤 2 的字节结果的 MD5：


> MD5(MD5('Password1')+'WebAPI')


结果作为字节：


> [ 144,75,168,236,177,98,115,210,240,174,156,59,138,2,55,82 ]


结果作为 HEX 字符串：


> 904ba8ecb16273d2f0ae9c3b8a023752


4。结合来自步骤 3 的密码字节散列以及 SRV_RAND：


> MD5(MD5('Password1')+'WebAPI')+ [ 115,0,125,199,24,71,71,206,15,124,152,81,110,241,200,81 ]


在此[ 115,0,125,199,24,71,71,206,15,124,152,81,110,241,200,81 ]是转换成字节数组的HEX 字符串73007dc7184747ce0f7c98516ef1c851：


> srv_rand = 73007dc7184747ce0f7c98516ef1c851 = [ 115,0,125,199,24,71,71,206,15,124,152,81,110,241,200,81 ]


结合的最终结果（两个字节数组被简单地结合起来）：


> [ 144,75,168,236,177,98,115,210,240,174,156,59,138,2,55,82,115,0,125,199,24,71,71,206,15,124,152,81,110,241,200,81 ]


5. 构成最终SRV_RAND_ANSWER以作为来自步骤4的字节结果的MD5对服务器响应：


> MD5(MD5(MD5('Password1')+'WebAPI') + [ 115,0,125,199,24,71,71,206,15,124,152,81,110,241,200,81 ]


结果作为字节：


> [ 119,254,81,130,127,127,166,157,216,15,190,201,170,51,241,187 ]


结果被转换成 HEX 表现：


> 77fe51827f7fa69dd80fbec9aa33f1bb


响应字符串中的最终参数如下：


> srv_rand_answer=77fe51827f7fa69dd80fbec9aa33f1bb


身份验证结束


通过交易服务器验证客户之后，访问服务器发送响应：


```
{

 "retcode" : "0 Done",

 "version_access" : "1290",

 "version_trade" : "1290",

 "cli_rand_answer" : "8b67609a264764a528ec8c709ab71df2"

}



LLLLKKKKFAUTH_ANSWER|RETCODE=0 Done|CLI_RAND_ANSWER=8b67609a264764a528ec8c709ab71df2|CRYPT_RAND=HEX_256_bytes\r\n
```


在命令中传递以下用"|"字符隔开的参数：


- retcode — 响应代码及其字符串描述。0 意味着成功完成。

- version_access — 用于连接至交易服务器的访问服务器的版本；

- version_trade — 交易服务器版本；

- cli_rand_answer — 随机序列的MD5散列和密码散列。序列是Hex格式，大小为16个字节。


为了授权客户服务器，使用了与服务器上客户授权所用相同的算法。在cli_rand_answer参数中，服务器发送客户的随机序列的MD5散列和密码散列。


构成cli_rand_answer的示例：


```
password hash=MD5(MD5('Password1')+'WebAPI')=904ba8ecb16273d2f0ae9c3b8a023752

cli_rand=4db98fec17aab4dc5a240bdc659e8395

//--- 计算公式：cli_rand_answer=MD5(password hash+CLI_RAND)

cli_rand_answer=MD5(904ba8ecb16273d2f0ae9c3b8a023752+4db98fec17aab4dc5a240bdc659e8395)=1732021b1b332f51660b2b315eed3a35
```

转义字符


使用特殊字符 =（等于）、|（竖线）、\（反斜杠）和换行键时，其之前应该使用转义字符 \（反斜杠）。转义字符应该只用于命令，其参数和参数值（在第一个换行字符之前）。附加主体可以用任何形式表示。


> 如果 \（反斜杠）后面不跟任何上述特殊字符，其按原样处理。


下表显示了如何在交易服务器上处理转义字符的示例。


| 已发送字符 | 服务器识别的字符 |
| --- | --- |
| \= | = |
| \\| | \| |
| \（换行键） | （换行键） |
| \\ | \ |


示例：


```
//--- As a result of the request, the user name will be changed to John|Smith

/api/user/update?login=1000&name=John\|Smith

//--- when requesting a group using the /api/group/get command, the path to the group can be specified in two ways:

/api/group/get?group=manager\administrators

/api/group/get?group==manager\\administrators
```


使用PHP时，请求的JSON主体中的正斜杠必须转义两次：


> $result=$request->Post('/api/user/update?login=10098', '{ "Group" : "demo\\\\demoforex" }');


如果您在请求URL中使用空格，请将其替换为%20。示例：


```
GET /api/user/add?name=John%20Smith&pass_main=1Rp0#jahr&pass_investor=lOp3$qnm&group=demo&leverage=100
```
